Useful widgets pt. 2

Progressbar

Progressbar to widżet paska ładowania (na zdjęciu tego nie widać, ale pasek się przesuwa).


									from tkinter import *
									from tkinter.ttk import *
									import time

									window = Tk()
									myProgressbar = Progressbar(window, orient=HORIZONTAL, length=100)

									def bar():
										myProgressbar["value"] = 20
										window.update_idletasks()
										time.sleep(1)

										myProgressbar["value"] = 40
										window.update_idletasks()
										time.sleep(1)

										myProgressbar["value"] = 50
										window.update_idletasks()
										time.sleep(1)

										myProgressbar["value"] = 60
										window.update_idletasks()
										time.sleep(1)

										myProgressbar["value"] = 80
										window.update_idletasks()
										time.sleep(1)
										myProgressbar["value"] = 100

									myProgressbar.pack()
									Button(window, text="Start", command=bar).pack()

									window.mainloop()

Zamiast komend time.sleep(1) możemy wstawić własne funkcje, po których wykonaniu pasek ma się przesunąć dalej.

TreeApp

Treeapp służy do tworzenia aplikacji "w stylu drzewa".


									import tkinter
									import tkinter.ttk as ttk
									window = tkinter.Tk()
									tkinter.Label(window, text="Treeview app").pack()
									myTreeview = ttk.Treeview(window)
									myTreeview.pack()
									myTreeview.insert("", "0", "item1", text="Wonsz")
									myTreeview.insert("item1", "1", "item2", text="Python")
									myTreeview.insert("item2", "end", "Programowanie", text="Programowanie")
									myTreeview.insert("item2", "end", "Biblioteki", text="Biblioteki")
									window.mainloop()

Za pomocą tego widżetu można też tworzyć tabele.


									import tkinter
									import tkinter.ttk as ttk
									window = tkinter.Tk()

									myTreeview = ttk.Treeview(window)
									myTreeview["columns"] = ["Python", "Java"]
									myTreeview.column("#0", width=50, anchor=tkinter.W)
									myTreeview.column("Python", width=120, anchor=tkinter.W)
									myTreeview.column("Java", width=120, minwidth=25)

									myTreeview.heading("#0", text="Numer", anchor=tkinter.W)
									myTreeview.heading("Python", text="Python", anchor=tkinter.W)
									myTreeview.heading("Java", text="Java", anchor=tkinter.W)

									myTreeview.insert(parent="", index="end", iid=0, text="1", values=["slow", "fast"])
									myTreeview.insert(parent="", index="end", iid=1, text="2", values=["easy to learn", "hard to learn"])
									myTreeview.insert(parent="", index="end", iid=2, text="3", values=["dinamically typed", "statically typed"])

									myTreeview.pack()
									window.mainloop()

Spinbox

Spinbox to okienko z liczbą, którą możesz zmniejszać i zwiększać, klikając na strzałki po prawej stronie (można ją też zmieniać, po prostu klikając w okienko i wpisując inną liczbę). Wybraną wartość pobieramy do zmiennej w taki sam sposób, jak przy komendzie Enter.


									import tkinter
									window = tkinter.Tk()
									mySpinbox = tkinter.Spinbox(window, from_=0, to=10)
									mySpinbox.pack()
									window.mainloop()

Scrollbar i Listbox

Scrollbar to pasek przewijania. Listbox pozwala na wyświetlenie listy elementów.


									import tkinter
									window = tkinter.Tk()
									myScrollbar = tkinter.Scrollbar(window)
									myScrollbar.pack(side=tkinter.RIGHT, fill=tkinter.Y)
									myListbox = tkinter.Listbox(window, yscrollcommand=myScrollbar.set)

									lista = ["a", "b", "c", "d", "e", "f", "g", "h", "i", "j", "k", "l", "m", "n", "o", "p", "r", "s", "t", "u", "w", "y", "z"]
									for x in lista:
										myListbox.insert(tkinter.END, str(x))
										
									myListbox.pack(side=tkinter.LEFT, fill=tkinter.BOTH)
									myScrollbar.config(command=myListbox.yview)
									window.mainloop()

Scale

Scale to skala, z której użytkownik może wybrać liczbę, przesuwając suwak. Może być pionowa lub pozioma. Wybraną wartość pobieramy do zmiennej w taki sam sposób, jak przy komendzie Enter.


									import tkinter
									window = tkinter.Tk()
									myScale1 = tkinter.Scale(window, from_=0, to=42)
									myScale1.pack()
									myScale2 = tkinter.Scale(window, from_=0, to=200, orient=tkinter.HORIZONTAL)
									myScale2.pack()
									window.mainloop()
