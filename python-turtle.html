Turtle

Biblioteka Turtle pozwala tworzyć różne rysunki i jest przeznaczona wyłącznie do nauki.


									from turtle import *

									#Komenda forward() powoduje poruszenie się żółwia do przodu. Można ją też zapisać jako fd().
									forward(100)

									#Komenda backward() powoduje poruszenie się żółwia do tyłu. Można ją też zapisać jako bk() lub back().
									backward(200)

									#Komenda right() powoduje poruszenie się żółwia w prawo. Można ją też zapisać jako rt().
									right(90)
									forward(100)

									#Komenda left() powoduje poruszenie się żółwia w lewo. Można ją też zapisać jako lt().
									left(90)
									forward(100)

									#Komenda circle() powoduje narysowanie koła przez żółwia.
									circle(100)

									#Komenda penup() podnosi pisak, a pendown() go opuszcza (kiedy jest podniesiony, żółw nie może rysować).
									#Komenda dot() powoduje postawienie przez żółwia kropki.
									for x in range(4):
											left(90)
											dot(20)
											penup()
											forward(100)
											pendown()

									#Komenda speed() ustawia prędkość żółwia (najszybsza: 0, szybka: 10, 
									#normalna: 6, wolna: 3, najwolniejsza: 1).
									speed(10)

									#Komenda undo() cofa ostatnią akcję żółwia.

									#Komenda shape() ustawia kształt żółwia (domyślnie jest to strzałka, ale może to być np. żółw).
									shape("turtle")
									#Inne możliwe kształty to: arrow, turtle, circle, square, triangle i classic.

									#Po użyciu komendy stamp(), żółw idąc, będzie stawiał naklejki wyglądające jak jego obecny shape.
									for x in range(4):
											penup()
											forward(120)
											pendown()
											stamp()
											if x < 2:
													left(90)
											else:
													right(90)

									#Komenda clearstamps() kasuje tyle stampów, ile podamy w nawiasie (ta komenda liczy od 1). 

									#Komenda goto() przeniesie żółwia na współrzędne, które wprowadzisz.
									goto(0, 0)
									forward(100)

									#Komenda position() pobiera aktualne położenie żółwia.
									print(position())

									#Komenda pencolor() ustawia kolor pisaka.
									pencolor("red")
									left(90)
									forward(100)

									#Komenda fillcolor() ustawia kolor wypełnienia kształtu. 
									#W miejscu, w którym chcemy zacząć wypełniać, używamy komendy begin_fill(), 
									#a w miejscu, w którym chcemy skończyć - end_fill().
									fillcolor("red")
									begin_fill()
									for x in range(4):
											forward(100); right(90)
									end_fill()

									#Komenda home() ustawia żółwia na współrzędnych startowych (0, 0).
									home()
