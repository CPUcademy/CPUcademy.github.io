More about pointers

Strings and chars

Every character in a string is a char, which basically means a string is an array of chars, and because of this, strings can be indexed, e.g., text[2]. We can also create an array of chars like this: char a[] = "abcdefg";. The \0 symbol means the end of a string. We can edit strings just like we edit arrays: a[2] = 'z';.


string text = "abcdefg"; // a b c d e f h \0
string text2 = "abcd\0efg"; // a b c d

The c_str() command turns a string into a string of a type that appears in the C language (it is literally an array of chars).


string text = "abcdefg";
const char *pointer = text.c_str();
cout << pointer << endl;
while(*pointer) // it will execute till encountering \0
	cout << *pointer++ << endl;

Of course, we can create dynamic string or char arrays (e.g., char *text = new char[20];) and the normal ones, too (string texts[10];). When working with string arrays, we can reference to the individual string's indexes just like we reference to objects in multidimensional arrays: texts[0][3] (this will get us to the fourth letter of the first string).

Functions

In the examples below, you can see the differences between reference variables and pointers applied in functions. Run them and analyze the results.


#include <iostream>
using namespace std;

void multiply1(int &x, int n)
{
	// int &x = a;
	x *= n;
}

void multiply2(int *x, int n)
{
	// int *x = &b;
	*x *= n;
}

int main()
{
	int a = 10;
	int b = 10;
	multiply1(a, 5);
	multiply2(&b, 5);
	
	cout << a << endl;
	cout << b << endl;
	
	return 0;
}
									

#include <iostream>
using namespace std;

int multiply(int *x, int n)
{
	*x *= n;
	return *x;
}

int main()
{
	int c = 10;
	int d = multiply(&c, 5);
	cout << c << endl;
	d = 20;
	cout << c << endl;
	cout << d << endl;
	
	return 0;
}
									

#include <iostream>
using namespace std;

int *multiply(int *x, int n) // a function can also be a pointer
{
	*x *= n;
	return x;
}

int main()
{
	int e = 10;
	int *f = multiply(&e, 5);
	cout << e << endl;
	*f = 20; // it also changes the "e" variable
	cout << e << endl;
	cout << *f << endl;
	
	return 0;
}
									

#include <iostream>
using namespace std;

void multiplyTab(int *t, int n, int size)
{
	// int *t = a;
	while(size--)
		t[size] *= n;
}

int main()
{
	int a[20];
	
	cout << "Size of an int variable: " << sizeof(int) << endl;
	cout << sizeof(a) << endl; // 20 * 4 bytes = 80 bytes
	
	for(int i = 0; i < sizeof(a) / sizeof(int); i++)
	{
		a[i] = i;
		cout << "a[" << i << "] = " << a[i] << endl;
	}
	
	cout << endl;
	multiplyTab(a, 5, sizeof(a) / sizeof(int));
	
	for(int i = 0; i < sizeof(a) / sizeof(int); i++)
		cout << "a[" << i << "] = " << a[i] << endl;
		
	return 0;
}
			

#include <iostream>
using namespace std;

int charArraySize(char *a)
{
	int length = 0;
	while(a[length++]); // it will increment the "length" variable until it reaches the end of the "a" array
	return length;
}

int main()
{
	cout << b << endl;
	char array[5] = "lala"; // l a l a \0

}

The example below is very complicated, and I show it just so you know that something like this exists.


#include <iostream>
using namespace std;

int name_of_the_function(){}

double * name_of_the_function_2(int a, double *b) // this function returns a pointer
{
	cout << a << endl;
	return b;
}

// int & name_of_the_function() - this function would return a reference

int function(double *(*pointer_on_the_function_2)(int, double *))
{
	double *pointer;
	int a = 5;
	*pointer_on_the_function_2(a, pointer);
}

int main()
{
	cout << b << endl;
	int a;
	int (*pointer_on_the_function)(); // a pointer that contains an address of a function
	pointer_on_the_function = &name_of_the_function;
	
	double *(*pointer_on_the_function_2)(int, double *); // the first asterisk indicates that the function 
	// returns a pointer (we defined it earlier by writing this sign before the name of the function in its definition)
	pointer_on_the_function_2 = &name_of_the_function_2;
	
	double *p = NULL;
	function(pointer_on_the_function_2);
	
	return 0;
}
