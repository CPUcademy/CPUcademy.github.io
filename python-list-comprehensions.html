List comprehensions

Wstęp


									#Trzy sposoby na zapisanie dziesięciu liczb do listy
									#(wszystkie te sposoby działają też dla innych typów danych).
									
									#Sposób 1:
									lista1 = []
									for x in range(10):
											lista1.append(10)
									print(lista1)

									#Sposób 2:
									lista2 = [10]*10
									print(lista2)

									#Sposób 3:
									lista3 = [10 for x in range(10)]
									print(lista3)

Wyrażenia Listowe

Wyrażenia listowe zapewniają prosty sposób na tworzenie list w oparciu o jakąś sekwencję lub inną listę, którą można przeiterować.


									#Sposób na zapisanie znaków stringa jako elementów listy.
									lista = [litera for litera in "abcd"]
									print(lista)

									#Wydzielając znaki ze stringa można je zamienić w liczby lub kolekcje (oprócz słownika).
									lista1 = [int(znak) for znak in "0123456"]
									print(lista1)

									#Komenda split() służy pobierania danych do kilku zmiennych po spacji 
									#(wspominałem już o niej na lekcji o przydatnych komendach).
									x, y, z = [int(x) for x in input("Podaj trzy liczby oddzielone spacjami: ").split()]
									print(x+y+z)

									#Za pomocą wyrażeń listowych można też zapisywać listy dwuwymiarowe.
									lista2 = [["a", "b"] for x in range(10)]
									print(lista2)

									#W wyrażeniach listowych można używać różnych komend np. lower().
									lista3 = [x.lower() for x in "ABCDE"]
									print(lista3)
									

									#Jako obiekt do zwrócenia można podawać działania (mogą być na kilku zmiennych), 
									#a także listę, w której będzie wyrażenie listowe (i w tym wyrażeniu lista z wyrażeniem itd.).
									
									#Przykład 1:
									kwadratyLiczb = [x*x for x in range(5)]
									print(kwadratyLiczb)

									#Przykład 2:
									lista1 = [x*y for x, y in ([1, 2], [3, 4])]
									print(lista1)

									#Przykład 3:
									lista2 = [[x for x in range(3)] for x in range(10)]
									print(lista2)
									

									#Za pomocą wyrażeń listowych można tworzyć kolekcje.
									
									#Przykład 1:
									zbior = {x*x for x in range(10)}
									print(zbior)

									#Przykład 2:
									slownik = {x: x*x for x in range(10)}
									print(slownik)
									

									#W poniższym przykładzie zwróci się x*y. x będzie znaczyć a, a y będzie znaczyć a+1. Wystąpi dziesięć iteracji.
									lista1 = [x*y for x, y in ([a, a+1] for a in range(10))]
									print(lista1)

									#Poniżej stanie się to samo, co w poprzednim przykładzie, poza tym że wyniki zostaną zapisane w innych listach 
									#wewnętrznych razem z wynikiem (x+1)*(y+1), czyli (a+1)*(a+1)+1.
									lista2 = [[x*y, (x+1)*(y+1)] for x, y in ([a, a+1] for a in range(10))]
									print(lista2)

									#Do wyrażeń listowych można dodawać warunki.
									lista3 = [x*x for x in range(11) if x%2==0]
									print(lista3)

									#Komenda isdigit() sprawdza, czy wyznaczony element tekstu jest liczbą.
									napis = "jeden: 1, dwa: 2, trzy: 3, cztery: 4"
									lista4 = [int(znak) for znak in napis if znak.isdigit()]
									print(lista4)
